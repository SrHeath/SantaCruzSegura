CREATE DATABASE IF NOT EXISTS sistema_seguridad_vecinal CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE sistema_seguridad_vecinal;

-- 1. CATÁLOGOS BÁSICOS
CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre ENUM('vecino', 'admin_junta', 'policia', 'superadmin') NOT NULL UNIQUE
);

CREATE TABLE barrios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL UNIQUE,
    centro_lat DECIMAL(10,8),
    centro_lon DECIMAL(10,8)
);

CREATE TABLE tipos_delito (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL UNIQUE
);

-- 2. USUARIOS Y AUTENTICACIÓN (CU06, CU08, RF35, RF36, RF37)
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    rol_id INT NOT NULL,
    barrio_id INT,
    nombre_completo VARCHAR(150) NOT NULL,
    email VARCHAR(150) NOT NULL UNIQUE,
    telefono VARCHAR(20),
    password_hash VARCHAR(255) NOT NULL,
    estado ENUM('activo', 'inhabilitado', 'bloqueado') DEFAULT 'activo',
    intentos_fallidos INT DEFAULT 0,
    reportes_falsos_mes INT DEFAULT 0,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (rol_id) REFERENCES roles(id),
    FOREIGN KEY (barrio_id) REFERENCES barrios(id)
);

CREATE TABLE intentos_login (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(150),
    ip_origen VARCHAR(45),
    fecha_hora DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_email_fecha (email, fecha_hora)
);

-- 3. INCIDENTES Y REPORTES (CU01, CU02, RF07-RF11)
CREATE TABLE incidentes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tipo_delito_id INT NOT NULL,
    barrio_id INT NOT NULL,
    descripcion TEXT,
    latitud DECIMAL(10,8) NOT NULL,
    longitud DECIMAL(10,8) NOT NULL,
    fecha_hora_incidente DATETIME NOT NULL,
    estado ENUM('pendiente', 'validado', 'falso') DEFAULT 'pendiente',
    id_reportero INT, -- NULL para reportes anónimos
    id_validador INT,
    fecha_hora_validacion DATETIME,
    num_testigos INT DEFAULT 1,
    FOREIGN KEY (tipo_delito_id) REFERENCES tipos_delito(id),
    FOREIGN KEY (barrio_id) REFERENCES barrios(id),
    FOREIGN KEY (id_reportero) REFERENCES usuarios(id),
    FOREIGN KEY (id_validador) REFERENCES usuarios(id),
    -- Índice optimizado para detección de duplicados (RF08, RF09)
    INDEX idx_geo_tiempo (latitud, longitud, fecha_hora_incidente)
);

-- Registra testigos adicionales cuando se fusionan duplicados (RF10, RF11)
CREATE TABLE testigos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    incidente_id INT NOT NULL,
    id_usuario INT, -- NULL si es anónimo
    canal_origen ENUM('web', 'whatsapp', 'voz', 'telegram') DEFAULT 'web',
    fecha_reporte DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (incidente_id) REFERENCES incidentes(id) ON DELETE CASCADE,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id)
);

-- 4. ALERTAS Y SUSCRIPCIONES (CU05, CU09)
CREATE TABLE suscripciones_alertas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    barrio_id INT NOT NULL,
    canal_preferido ENUM('whatsapp', 'telegram', 'ambos') DEFAULT 'ambos',
    recibir_predictivas BOOLEAN DEFAULT TRUE,
    fecha_suscripcion DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_usuario_barrio (usuario_id, barrio_id),
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE,
    FOREIGN KEY (barrio_id) REFERENCES barrios(id)
);

-- 5. OPERATIVOS POLICIALES (CU04)
CREATE TABLE operativos_policiales (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_policia INT NOT NULL,
    barrio_id INT NOT NULL,
    fecha DATE NOT NULL,
    hora_inicio TIME,
    hora_fin TIME,
    num_agentes INT,
    zona_asignada VARCHAR(200),
    notas TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_policia) REFERENCES usuarios(id),
    FOREIGN KEY (barrio_id) REFERENCES barrios(id)
);

-- 6. MODELOS IA PREDICTIVA (CU07, RF33, RF15)
CREATE TABLE modelos_ia (
    id INT AUTO_INCREMENT PRIMARY KEY,
    version VARCHAR(50) NOT NULL,
    fecha_entrenamiento DATETIME DEFAULT CURRENT_TIMESTAMP,
    precision_obtenida DECIMAL(5,2),
    registros_usados INT,
    estado ENUM('activo', 'inactivo', 'descartado') DEFAULT 'activo',
    ruta_archivo VARCHAR(255)
);

-- 7. AUDITORÍA Y LOGS (RF39, RNF10, CU08)
CREATE TABLE logs_auditoria (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT,
    accion VARCHAR(100) NOT NULL,
    descripcion TEXT,
    ip_origen VARCHAR(45),
    fecha_hora DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE SET NULL,
    INDEX idx_accion_fecha (accion, fecha_hora)
);
